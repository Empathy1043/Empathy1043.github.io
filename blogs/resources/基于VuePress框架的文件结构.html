以下是为你设计的VuePress博客项目文件结构和配置方案，包含具体实现步骤和最佳实践建议：

# 一、推荐文件结构图

docs/
├── .vuepress/               # 核心配置目录
│   ├── public/             # 静态资源
│   │   ├── avatar.png      # 头像
│   │   ├── favicon.ico     # 网站图标
│   │   └── banner.jpg      # 首页横幅

│   │   └── 2024-ai-notes/

├── about/                  # 关于页面

# 二、核心配置文件 (config.js)

module.exports = {
  title: '知识探索者',  // 网站标题
  description: '终身学习者的数字花园', // 网站描述
  head: [
    ['link', { rel: 'icon', href: '/favicon.ico' }],
    ['meta', { name: 'theme-color', content: '#3eaf7c' }]
  ],
  
  themeConfig: {
    logo: '/avatar.png',
    nav: [
      { text: '首页', link: '/' },
      { text: '博客', link: '/blogs/' },
      { text: '分类', 
        items: [
          { text: '技术笔记', link: '/blogs/learning/' },
          { text: '个人成长', link: '/blogs/growth/' },
          { text: '资源合集', link: '/blogs/resources/' }
        ]
      },
      { text: '关于', link: '/about/' }
    ],
    
    // 自动生成侧边栏
    sidebar: {
      '/blogs/learning/': [
        {
          title: '技术学习',
          collapsable: false,
          children: [
            '2023-web-dev/',
            '2024-ai-notes/'
          ]
        }
      ],
      '/blogs/growth/': [
        {
          title: '成长轨迹',
          collapsable: false,
          children: [
            '2023-review',
            '2024-plan'
          ]
        }
      ]
    },
    
    // 社交链接
    socialLinks: [
      { icon: 'github', link: 'https://github.com/yourusername' }
    ],
    
    // 文档更新时间
    lastUpdated: true,
    
    // 搜索功能
    search: true,
    searchMaxSuggestions: 10
  },
  
  // Markdown扩展配置
  markdown: {
    lineNumbers: true,  // 代码行号
    extractHeaders: ['h2', 'h3', 'h4']  // 自动生成目录的标题级别
  },
  
  // 插件配置
  plugins: [
    ['@vuepress/plugin-search'],
    ['@vuepress/plugin-google-analytics', { id: 'G-XXXXXXX' }],
    ['@vuepress/plugin-medium-zoom']
  ]
}
